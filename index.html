-- Run this script in your Supabase SQL Editor:
CREATE TABLE site_settings (
    id BIGINT PRIMARY KEY,
    project_name TEXT,
    offer_title TEXT,
    offer_start_date DATE,
    offer_end_date DATE,
    main_road_width TEXT,
    sub_road_width TEXT,
    contact_phone TEXT
);

CREATE TABLE plots_info (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    plot_number TEXT UNIQUE,
    plot_size TEXT,
    facing TEXT,
    status TEXT, -- Available, Booked, Sold
    price_per_katha TEXT
);

CREATE TABLE banners_info (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    banner_title TEXT,
    image_url TEXT,
    link_url TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
